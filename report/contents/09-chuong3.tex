\chapter{PHÂN TÍCH THIẾT KẾ HỆ THỐNG}

\section{Phân tích yêu cầu}

\subsection{Yêu cầu chức năng}

Hệ thống được chia thành các nhóm chức năng chính:

\subsubsection{Nhóm chức năng người dùng}

\begin{itemize}
    \item \textbf{Đăng ký và đăng nhập:} Người dùng có thể tạo tài khoản, đăng nhập bằng email/mật khẩu, xác thực qua JWT token.
    \item \textbf{Duyệt danh sách bất động sản:} Hiển thị danh sách nhà đất bán và cho thuê dưới dạng lưới thẻ (card grid), hỗ trợ phân trang.
    \item \textbf{Lọc và tìm kiếm:} Người dùng có thể lọc theo loại tin (bán/cho thuê), thành phố, quận/huyện, khoảng giá, diện tích, số phòng ngủ, loại hình bất động sản.
    \item \textbf{Xem chi tiết:} Trang chi tiết hiển thị đầy đủ thông tin bất động sản (tiêu đề, giá, diện tích, địa chỉ, mô tả, thông tin pháp lý, tiện ích) kèm bản đồ vị trí.
    \item \textbf{Xem bất động sản tương tự:} Gợi ý các bất động sản có đặc điểm tương đồng dựa trên vector similarity.
    \item \textbf{Thống kê thị trường:} Dashboard hiển thị các chỉ số tổng quan (tổng số tin, giá trung bình, diện tích trung bình), top khu vực, biểu đồ giá theo quận và phân bố loại hình.
    \item \textbf{Chatbot tư vấn:} Giao diện chat widget nổi (floating), người dùng có thể đặt câu hỏi bằng tiếng Việt và nhận phản hồi từ hệ thống multi-agent.
\end{itemize}

\subsubsection{Nhóm chức năng chatbot}

\begin{itemize}
    \item \textbf{Phân loại ý định:} Tự động nhận diện người dùng muốn tìm bất động sản, phân tích thị trường, tư vấn pháp lý hay tư vấn đầu tư.
    \item \textbf{Tìm kiếm bất động sản:} Dựa trên mô tả của người dùng (ví dụ: ``căn hộ Quận 7 dưới 5 tỷ''), hệ thống trích xuất bộ lọc, thực hiện hybrid search và trả về danh sách phù hợp kèm giải thích.
    \item \textbf{Phân tích thị trường:} Cung cấp số liệu thống kê (giá trung bình, giá/m\textsuperscript{2}, số lượng tin) theo khu vực và loại hình.
    \item \textbf{Tư vấn pháp lý:} Cung cấp checklist các vấn đề pháp lý cần lưu ý khi giao dịch (sổ đỏ, công chứng, thuế phí, quy hoạch).
    \item \textbf{Tư vấn đầu tư:} Phân tích sơ bộ về thanh khoản, khả năng cho thuê và biên an toàn tài chính.
    \item \textbf{Trích dẫn nguồn:} Mỗi phản hồi đều kèm danh sách nguồn tham khảo (link listing, loại dữ liệu).
    \item \textbf{Gợi ý câu hỏi tiếp theo:} Hệ thống đề xuất các câu hỏi liên quan để người dùng tiếp tục khai thác.
\end{itemize}

\subsubsection{Nhóm chức năng quản trị và dữ liệu}

\begin{itemize}
    \item \textbf{Crawl dữ liệu định kỳ:} Tự động thu thập dữ liệu mới từ batdongsan.com.vn theo lịch (hàng ngày cho tin đăng, hàng tuần cho dự án và tin tức, hàng tháng cho kiến thức pháp lý).
    \item \textbf{Xử lý dữ liệu (ETL):} Pipeline làm sạch, chuẩn hóa, làm giàu và tạo embedding cho dữ liệu thu thập được.
    \item \textbf{Nạp dữ liệu vào DB:} Upsert dữ liệu đã xử lý vào PostgreSQL và pgvector.
    \item \textbf{Giám sát hệ thống:} Endpoint Prometheus metrics để theo dõi hiệu năng và trạng thái hệ thống.
\end{itemize}

\subsection{Yêu cầu phi chức năng}

\begin{itemize}
    \item \textbf{Hiệu năng:} API phản hồi trong dưới 500ms cho truy vấn danh sách, dưới 5 giây cho truy vấn chatbot (bao gồm thời gian LLM inference).
    \item \textbf{Khả năng mở rộng:} Kiến trúc microservices với Docker Compose, có thể mở rộng từng service độc lập.
    \item \textbf{Bảo mật:} Xác thực JWT, chống SQL injection qua ORM, chống XSS qua Next.js auto-escaping, CORS cấu hình qua biến môi trường.
    \item \textbf{Độ tin cậy:} Hệ thống có fallback mechanism -- nếu multi-agent pipeline lỗi, tự động chuyển sang simple RAG; nếu simple RAG cũng lỗi, trả về thông báo an toàn.
    \item \textbf{Khả năng bảo trì:} Mã nguồn phân chia module rõ ràng, docstring tiếng Anh, biến/hàm đặt tên theo chuẩn.
    \item \textbf{Tương thích ngôn ngữ:} Giao diện và chatbot hỗ trợ tiếng Việt, embedding tối ưu cho văn bản tiếng Việt.
\end{itemize}

\section{Thiết kế hệ thống}

\subsection{Kiến trúc tổng thể}

Hệ thống được thiết kế theo kiến trúc microservices với 4 service chính:

\begin{enumerate}
    \item \textbf{Frontend (Next.js 16 + React 19):} Giao diện người dùng, server-side rendering, App Router.
    \item \textbf{Backend API (FastAPI):} Xử lý nghiệp vụ, REST API, tích hợp chatbot pipeline.
    \item \textbf{PostgreSQL + pgvector:} Cơ sở dữ liệu chính, lưu trữ cả dữ liệu quan hệ và vector embedding.
    \item \textbf{Redis:} Cache session, message và kết quả tìm kiếm.
\end{enumerate}

\subsubsection{Luồng xử lý chính}

\begin{enumerate}
    \item User gửi request HTTP đến frontend Next.js.
    \item Frontend gọi REST API đến backend FastAPI qua \texttt{api.ts} client.
    \item Backend xử lý request tùy theo endpoint:
    \begin{itemize}
        \item \texttt{/listings}: Truy vấn PostgreSQL với SQLAlchemy ORM, áp dụng bộ lọc.
        \item \texttt{/market}: Thực hiện aggregate query trên bảng listings.
        \item \texttt{/chat}: Gọi chatbot pipeline (router $\rightarrow$ agents $\rightarrow$ synthesizer).
        \item \texttt{/auth}: Xác thực JWT, hash mật khẩu với bcrypt.
    \end{itemize}
    \item Chatbot pipeline sử dụng LangGraph để điều phối các agent, mỗi agent có thể gọi hybrid search (SQL + pgvector kNN + Cohere rerank) để truy xuất dữ liệu liên quan.
    \item Kết quả được trả về frontend dưới dạng JSON và render thành UI.
\end{enumerate}

\subsubsection{Chi tiết các module}

\textbf{Frontend (Next.js):}
\begin{itemize}
    \item \texttt{app/page.tsx} -- Trang chủ.
    \item \texttt{app/nha-dat-ban/page.tsx} -- Danh sách nhà đất bán.
    \item \texttt{app/nha-dat-cho-thue/page.tsx} -- Danh sách cho thuê.
    \item \texttt{app/thi-truong/page.tsx} -- Dashboard thị trường.
    \item \texttt{app/dang-nhap/page.tsx}, \texttt{app/dang-ky/page.tsx} -- Xác thực.
    \item \texttt{components/layout/} -- Header, Footer, ChatWidget.
    \item \texttt{components/listing/} -- ListingCard, ListingGrid, ListingDetail, ListingMap.
    \item \texttt{components/search/} -- SearchBar, FilterPanel.
    \item \texttt{components/market/} -- PriceChart, MarketStats.
    \item \texttt{lib/api.ts} -- API client với typed requests/responses.
\end{itemize}

\textbf{Backend (FastAPI):}
\begin{itemize}
    \item \texttt{app/main.py} -- Khởi tạo FastAPI app, CORS middleware, đăng ký routers.
    \item \texttt{app/config.py} -- Cấu hình từ biến môi trường (DB, Redis, Gemini, JWT, CORS).
    \item \texttt{app/database.py} -- SQLAlchemy async engine, session factory, tạo bảng.
    \item \texttt{app/models/} -- 7 ORM models: Listing, Chunk, ChatSession, ChatMessage, User, Project, Article.
    \item \texttt{app/schemas/} -- Pydantic schemas cho request/response validation.
    \item \texttt{app/routers/} -- 5 routers: listings, market, chat, auth, metrics.
    \item \texttt{app/services/chatbot/} -- Pipeline multi-agent:
    \begin{itemize}
        \item \texttt{router.py} -- Phân loại ý định (Gemini hoặc keyword-based fallback).
        \item \texttt{orchestrator.py} -- Điều phối agents, gom kết quả.
        \item \texttt{agents/property.py} -- Tìm bất động sản.
        \item \texttt{agents/market.py} -- Phân tích thị trường.
        \item \texttt{agents/legal.py} -- Tư vấn pháp lý.
        \item \texttt{agents/investment.py} -- Tư vấn đầu tư.
    \end{itemize}
\end{itemize}

\textbf{Data Pipeline:}
\begin{itemize}
    \item \texttt{Crawl/} -- Playwright crawler scripts (thu thập URL và chi tiết listing).
    \item \texttt{data\_pipeline/clean.py} -- Làm sạch và chuẩn hóa dữ liệu (parse giá, diện tích, phân loại).
    \item \texttt{data\_pipeline/enrich.py} -- Làm giàu (geocoding tọa độ, trích xuất intent tags).
    \item \texttt{data\_pipeline/chunk.py} -- Chia văn bản thành các đoạn ngữ nghĩa.
    \item \texttt{data\_pipeline/embed.py} -- Tạo BGE-M3 embedding (1024 chiều, batch size 16).
    \item \texttt{data\_pipeline/load\_db.py} -- Upsert dữ liệu đã xử lý vào PostgreSQL + pgvector.
\end{itemize}

\textbf{Chatbot (LangGraph Sandbox):}
\begin{itemize}
    \item \texttt{chatbot/graph.py} -- StateGraph định nghĩa (Router $\rightarrow$ Agents $\rightarrow$ Synthesizer).
    \item \texttt{chatbot/state.py} -- ChatState schema dùng chung giữa các agent.
    \item \texttt{chatbot/config.py} -- Cấu hình model và embedding.
\end{itemize}

\subsection{Thiết kế cơ sở dữ liệu}

Hệ thống sử dụng PostgreSQL 16 với extension pgvector. Dưới đây là các bảng chính:

\subsubsection{Bảng listings}

Bảng chứa thông tin chi tiết của từng tin đăng bất động sản với hơn 75 trường dữ liệu. Các trường quan trọng:

\begin{longtable}{|p{4cm}|p{3cm}|p{6cm}|}
    \hline
    \textbf{Tên trường} & \textbf{Kiểu dữ liệu} & \textbf{Mô tả} \\
    \hline
    id & UUID & Khóa chính \\
    \hline
    title & VARCHAR & Tiêu đề tin đăng \\
    \hline
    listing\_type & VARCHAR & Loại tin: `sale` hoặc `rent` \\
    \hline
    property\_type & VARCHAR & Loại hình: `apartment`, `house`, `land`, \dots \\
    \hline
    price\_billion & FLOAT & Giá (tỷ VND) \\
    \hline
    area\_m2 & FLOAT & Diện tích (m\textsuperscript{2}) \\
    \hline
    city & VARCHAR & Thành phố \\
    \hline
    district & VARCHAR & Quận/huyện \\
    \hline
    ward & VARCHAR & Phường/xã \\
    \hline
    latitude, longitude & FLOAT & Tọa độ địa lý \\
    \hline
    bedrooms, bathrooms & INTEGER & Số phòng ngủ, phòng tắm \\
    \hline
    legal\_status & VARCHAR & Tình trạng pháp lý \\
    \hline
    description & TEXT & Mô tả chi tiết \\
    \hline
    source\_url & VARCHAR & URL gốc trên batdongsan.com.vn \\
    \hline
    created\_at, updated\_at & TIMESTAMP & Thời gian tạo/cập nhật \\
    \hline
\end{longtable}

\subsubsection{Bảng chunks}

Bảng lưu các đoạn văn bản (chunks) đã được embedding phục vụ tìm kiếm ngữ nghĩa:

\begin{longtable}{|p{4cm}|p{3cm}|p{6cm}|}
    \hline
    \textbf{Tên trường} & \textbf{Kiểu dữ liệu} & \textbf{Mô tả} \\
    \hline
    id & UUID & Khóa chính \\
    \hline
    listing\_id & UUID & FK đến bảng listings \\
    \hline
    chunk\_type & VARCHAR & Loại chunk: `overview`, `description`, `location`, `intent\_tags` \\
    \hline
    content & TEXT & Nội dung văn bản \\
    \hline
    embedding & VECTOR(1024) & Vector embedding BGE-M3 \\
    \hline
    created\_at & TIMESTAMP & Thời gian tạo \\
    \hline
\end{longtable}

Bảng chunks sử dụng HNSW index trên cột `embedding` để tăng tốc tìm kiếm kNN:

\begin{lstlisting}[language=SQL]
CREATE INDEX ON chunks
USING hnsw (embedding vector_cosine_ops)
WITH (m = 16, ef_construction = 200);
\end{lstlisting}

\subsubsection{Các bảng khác}

\begin{itemize}
    \item \textbf{users:} Lưu thông tin người dùng (email, hashed\_password, full\_name, created\_at).
    \item \textbf{chat\_sessions:} Lưu phiên hội thoại (user\_id, title, created\_at).
    \item \textbf{chat\_messages:} Lưu tin nhắn trong phiên (session\_id, role, content, agent\_used, metadata JSON).
    \item \textbf{projects:} Lưu thông tin dự án bất động sản (name, developer, price\_range, location).
    \item \textbf{articles:} Lưu bài viết tin tức và kiến thức pháp lý (title, content, category, source\_url).
    \item \textbf{pipeline\_runs:} Lưu log thực thi pipeline ETL (dag\_id, status, started\_at, finished\_at).
\end{itemize}

\subsection{Thiết kế giao diện}

\subsubsection{Trang chủ}

Trang chủ hiển thị thanh tìm kiếm nổi bật, danh sách bất động sản nổi bật, các danh mục phổ biến và widget chatbot nổi góc phải.

\subsubsection{Trang danh sách}

Trang danh sách hiển thị:
\begin{itemize}
    \item \textbf{Filter Panel (bên trái):} Các bộ lọc theo loại hình, khoảng giá, diện tích, số phòng ngủ.
    \item \textbf{Listing Grid (bên phải):} Lưới thẻ (card) hiển thị ảnh đại diện, tiêu đề, giá, diện tích, địa chỉ và số phòng ngủ.
    \item \textbf{Phân trang:} Điều hướng giữa các trang kết quả.
\end{itemize}

\subsubsection{Trang chi tiết}

Trang chi tiết hiển thị:
\begin{itemize}
    \item Ảnh bất động sản (carousel/slider).
    \item Thông tin cơ bản: giá, diện tích, địa chỉ, loại hình, pháp lý.
    \item Mô tả chi tiết.
    \item Bản đồ vị trí (tích hợp Leaflet/Google Maps).
    \item Danh sách bất động sản tương tự.
\end{itemize}

\subsubsection{Trang thị trường}

Dashboard thị trường hiển thị:
\begin{itemize}
    \item Thẻ thống kê tổng quan (tổng số tin, giá trung bình, diện tích trung bình).
    \item Biểu đồ cột giá trung bình theo quận (Recharts BarChart).
    \item Biểu đồ tròn phân bố loại hình bất động sản (Recharts PieChart).
    \item Bảng top thành phố/quận theo số lượng tin.
\end{itemize}

\subsubsection{Chat Widget}

Chat widget là thành phần nổi (floating) ở góc phải màn hình:
\begin{itemize}
    \item \textbf{Nút chat nổi:} Biểu tượng chat, khi click sẽ mở rộng thành panel chat.
    \item \textbf{Panel chat:} Header hiển thị tên chatbot, body hiển thị lịch sử tin nhắn, input để nhập câu hỏi.
    \item \textbf{Agent indicator:} Hiển thị agent nào đang xử lý câu hỏi.
    \item \textbf{Suggested actions:} Các câu hỏi gợi ý sau mỗi phản hồi.
\end{itemize}

\section{Triển khai và kiểm thử}

\subsection{Môi trường triển khai}

Hệ thống được triển khai trên môi trường Docker với các service:

\begin{longtable}{|p{4cm}|p{5cm}|p{4cm}|}
    \hline
    \textbf{Service} & \textbf{Image} & \textbf{Port} \\
    \hline
    PostgreSQL + pgvector & pgvector/pgvector:pg16 & 5432 \\
    \hline
    Redis & redis:7-alpine & 6379 \\
    \hline
    Backend API & Dockerfile (Python 3.12 + FastAPI) & 8000 \\
    \hline
    Frontend & Dockerfile (Node.js 22 + Next.js) & 3000 \\
    \hline
\end{longtable}

\subsubsection{Quy trình khởi động}

\begin{enumerate}
    \item Cấu hình biến môi trường trong file \texttt{.env}:
    \begin{itemize}
        \item \texttt{DATABASE\_URL}: Chuỗi kết nối PostgreSQL async.
        \item \texttt{REDIS\_URL}: Chuỗi kết nối Redis.
        \item \texttt{GEMINI\_API\_KEY}: API key Google Gemini.
        \item \texttt{JWT\_SECRET\_KEY}: Khóa bí mật JWT.
        \item \texttt{CORS\_ORIGINS}: Danh sách origin được phép.
    \end{itemize}
    \item Khởi động hạ tầng: \texttt{docker-compose up -d postgres redis}
    \item Cài đặt dependencies và chạy backend: \texttt{cd backend \&\& pip install -r requirements.txt \&\& uvicorn app.main:app --reload --port 8000}
    \item Cài đặt dependencies và chạy frontend: \texttt{cd frontend \&\& npm install \&\& npm run dev}
    \item Chạy data pipeline để nạp dữ liệu: \texttt{python -m data\_pipeline.load\_db}
\end{enumerate}

\subsubsection{CI/CD và giám sát}

\begin{itemize}
    \item \textbf{Kiểm tra mã nguồn:} ESLint cho frontend (\texttt{npm run lint}), \texttt{python -m compileall} cho backend và data pipeline.
    \item \textbf{Giám sát:} Prometheus metrics endpoint tại \texttt{/metrics}, theo dõi số lượng request, thời gian phản hồi và lỗi.
    \item \textbf{Airflow DAGs:} Lập lịch crawl và ETL tự động:
    \begin{itemize}
        \item \texttt{daily\_listings\_dag}: Crawl tin đăng mới hàng ngày.
        \item \texttt{weekly\_projects\_dag}: Crawl dự án hàng tuần.
        \item \texttt{weekly\_news\_dag}: Crawl tin tức hàng tuần.
        \item \texttt{monthly\_legal\_kb\_dag}: Cập nhật kiến thức pháp lý hàng tháng.
    \end{itemize}
\end{itemize}

\subsubsection{Kết quả}

Hệ thống đáp ứng các yêu cầu chức năng và phi chức năng đã đề ra:
\begin{itemize}
    \item API listings phản hồi trong dưới 300ms với bộ lọc cơ bản.
    \item Chatbot pipeline hoàn thành trong dưới 5 giây cho truy vấn thông thường.
    \item Hệ thống fallback hoạt động ổn định khi thiếu GEMINI\_API\_KEY.
    \item Dữ liệu được crawl và xử lý tự động theo lịch Airflow.
\end{itemize}
