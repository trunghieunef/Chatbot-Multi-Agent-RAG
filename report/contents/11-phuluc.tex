\subsubsection*{Luồng dữ liệu từ Crawl đến Hiển thị}

\begin{enumerate}
    \item \textbf{Crawl (Playwright + Stealth):} Thu thập dữ liệu từ batdongsan.com.vn với 4 workers song song. Output: file CSV UTF-8 BOM, deduplicate theo product\_id. Thực hiện bởi 4 module crawler (sale, rent, projects, news).

    \item \textbf{Clean (data\_pipeline/clean.py):} Chuẩn hóa dữ liệu thô -- parse giá (``4,68 tỷ'' $\rightarrow$ price=4.68, unit=billion), parse diện tích (``75 m\textsuperscript{2}'' $\rightarrow$ area=75), phân loại property\_type, chuẩn hóa địa chỉ (tách ward, district, city).

    \item \textbf{Enrich (data\_pipeline/enrich.py):} Làm giàu dữ liệu với 2 tác vụ: (a) Geocoding -- gọi Nominatim/Goong API chuyển địa chỉ text $\rightarrow$ tọa độ (latitude, longitude); (b) Intent Tagging -- phân tích mô tả, gán nhãn nhu cầu (``gần trường học'', ``view đẹp'', ``an ninh'', ``pháp lý rõ'').

    \item \textbf{Chunk (data\_pipeline/chunk.py):} Chia mỗi listing thành 4 loại chunk ngữ nghĩa độc lập:
    \begin{itemize}
        \item \texttt{overview} -- Tổng quan: tiêu đề, giá, diện tích, khu vực, pháp lý, nội thất.
        \item \texttt{description} -- Mô tả chi tiết từ người đăng.
        \item \texttt{location} -- Vị trí: địa chỉ, quận/huyện, tỉnh/thành phố.
        \item \texttt{intent\_tags} -- Nhu cầu phù hợp (dựa trên intent tagging).
    \end{itemize}

    \item \textbf{Embed (data\_pipeline/embed.py):} Tạo vector embedding 1024 chiều cho từng chunk bằng BGE-M3 (BAAI/bge-m3). Sử dụng sentence-transformers với batch\_size=16, normalize=True, retry 3 lần exponential backoff.

    \item \textbf{Load DB (data\_pipeline/load\_db.py + ingestors/):} Upsert vào PostgreSQL qua asyncpg driver với batch processing (25--50 bản ghi/batch). Dữ liệu quan hệ $\rightarrow$ bảng listings; dữ liệu vector $\rightarrow$ bảng chunks với HNSW index (m=16, ef\_construction=64).

    \item \textbf{Serve (FastAPI + Next.js):} Backend API truy xuất dữ liệu qua hybrid search (SQL filtering + pgvector kNN + Cohere rerank + Redis cache). Frontend hiển thị qua Next.js App Router với SSR.

    \item \textbf{Schedule (Apache Airflow):} Toàn bộ pipeline từ bước 1--6 được lập lịch tự động qua 4 DAGs (daily, weekly$\times$2, monthly) với retry policy và alerting.
\end{enumerate}

\subsubsection*{Luồng xử lý Chatbot}

\begin{enumerate}
    \item \textbf{User gửi query} qua ChatWidget (Frontend).
    \item \textbf{Frontend gọi} \texttt{POST /api/v1/chat} (Backend).
    \item \textbf{Backend xác thực} (JWT hoặc anonymous), kiểm tra quota, lưu ChatMessage.
    \item \textbf{Router phân loại ý định} (keyword-based hoặc Gemini-based).
    \item \textbf{Orchestrator chọn agents} và gọi song song qua asyncio.gather().
    \item \textbf{Agent thực thi} hybrid search (SQL + pgvector kNN $\pm$ Cohere rerank $\pm$ Redis cache).
    \item \textbf{Kết quả tổng hợp} $\rightarrow$ AgentChatResponse (final\_response, sources, suggested\_actions).
    \item \textbf{Nếu Agent Service được kích hoạt}, Backend ủy thác sang Agent Service (LangGraph 8-node workflow) thay vì pipeline nội bộ.
    \item \textbf{Backend lưu} ChatMessage, AgentTrace và trả kết quả về Frontend.
    \item \textbf{Frontend hiển thị} phản hồi kèm sources và suggested actions.
\end{enumerate}

\textbf{Ví dụ 1: Tìm kiếm bất động sản}

\begin{quote}
\textbf{User:} ``Tôi muốn mua căn hộ chung cư ở Quận 7, giá dưới 5 tỷ, 2 phòng ngủ''

\textbf{Intent:} property\_search

\textbf{Agent sử dụng:} property\_search

\textbf{Bộ lọc trích xuất:} \texttt{\{listing\_type: "sale", property\_type: "apartment", district: "Quận 7", max\_price: 5, bedrooms: 2\}}

\textbf{Phản hồi:}
Dựa trên yêu cầu của bạn, tôi tìm thấy một số căn hộ phù hợp tại Quận 7:

\begin{enumerate}
    \item \textbf{Căn hộ chung cư ABC} -- 4,2 tỷ, 72m\textsuperscript{2}, 2PN, 2WC, Phường Tân Phong. View sông, nội thất đầy đủ, pháp lý sổ hồng.
    \item \textbf{Căn hộ Sunshine City} -- 4,8 tỷ, 78m\textsuperscript{2}, 2PN, 1WC, Phường Phú Thuận. Gần trường học, an ninh tốt, có hồ bơi.
    \item \textbf{Căn hộ Park View} -- 3,9 tỷ, 68m\textsuperscript{2}, 2PN, 1WC, Phường Tân Quy. Giá tốt nhất khu vực, gần siêu thị.
\end{enumerate}

\textbf{Nguồn tham khảo:} [Link BĐS 1] (độ liên quan: 0.95), [Link BĐS 2] (0.89), [Link BĐS 3] (0.82)

\textbf{Gợi ý tiếp theo:}
\begin{itemize}
    \item ``So sánh giá/m\textsuperscript{2} các căn hộ này''
    \item ``Thủ tục pháp lý khi mua chung cư cần những gì?''
    \item ``Khu vực Quận 7 có tiềm năng tăng giá không?''
\end{itemize}
\end{quote}

\vspace{0.5cm}

\textbf{Ví dụ 2: Tư vấn pháp lý + thị trường}

\begin{quote}
\textbf{User:} ``Thủ tục sang tên sổ đỏ mất bao lâu? Với lại giá đất ở Quận 9 hiện nay thế nào?''

\textbf{Intent:} mixed (legal\_advice + market\_analysis)
