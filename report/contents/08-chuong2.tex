\chapter{PHÂN TÍCH THIẾT KẾ HỆ THỐNG}

Chương này trình bày toàn bộ quá trình phân tích và thiết kế hệ thống. Nội dung bao gồm: phân tích yêu cầu chức năng và phi chức năng, thiết kế kiến trúc tổng thể theo mô hình dịch vụ vi mô, thiết kế chi tiết các mô-đun thành phần (giao diện người dùng, máy chủ dịch vụ, dịch vụ tác tử, đường ống dữ liệu), thiết kế cơ sở dữ liệu quan hệ kết hợp véc-tơ, thiết kế giao diện người dùng, quy trình triển khai với Docker và kết quả kiểm thử hệ thống.

\section{Phân tích yêu cầu}

\subsection{Yêu cầu chức năng}

Hệ thống phục vụ ba nhóm đối tượng chính: người dùng cuối (người mua, người thuê, nhà đầu tư), quản trị viên hệ thống, và bản thân hệ thống (các tác vụ tự động). Dưới đây là các nhóm chức năng được phân tích chi tiết.

\subsubsection{Nhóm chức năng dành cho người dùng cuối}

Đây là nhóm chức năng cốt lõi, quyết định trải nghiệm của người dùng khi tương tác với nền tảng:

\begin{enumerate}
    \item \textbf{Xác thực và quản lý tài khoản:}
    \begin{itemize}
        \item Đăng ký tài khoản mới với email và mật khẩu. Mật khẩu được hash bằng bcrypt trước khi lưu trữ.
        \item Đăng nhập bằng email/mật khẩu, hệ thống trả về JWT token có thời hạn 24 giờ. Token được lưu ở client (localStorage) và gửi kèm trong Authorization header cho các request tiếp theo.
        \item Người dùng chưa đăng nhập (anonymous) vẫn có thể sử dụng chatbot với giới hạn 20 lượt/ngày. Người dùng đã xác thực được nâng lên 200 lượt/ngày.
        \item Hồ sơ người dùng bao gồm: họ tên, email, số điện thoại, ảnh đại diện, quyền (is\_admin, is\_superuser).
    \end{itemize}

    \item \textbf{Duyệt và tìm kiếm bất động sản:}
    \begin{itemize}
        \item \textbf{Danh sách bất động sản bán:} Hiển thị dưới dạng lưới thẻ (card grid) với phân trang (pagination). Mỗi thẻ hiển thị: tiêu đề, giá, diện tích, địa chỉ (quận, thành phố), số phòng ngủ, loại tin (VIP/thường).
        \item \textbf{Danh sách bất động sản cho thuê:} Tương tự như trang bán nhưng dành riêng cho tin cho thuê.
        \item \textbf{Bộ lọc nâng cao:} Người dùng có thể lọc theo 10+ tiêu chí: loại tin (bán/cho thuê), loại hình bất động sản (căn hộ, nhà riêng, đất nền, biệt thự,\dots), thành phố, quận/huyện, khoảng giá (từ -- đến), diện tích (từ -- đến), số phòng ngủ, số phòng tắm, hướng nhà.
        \item \textbf{Tìm kiếm theo từ khóa:} Hỗ trợ tìm kiếm full-text trên các trường tiêu đề, mô tả, địa chỉ, quận, thành phố.
        \item \textbf{Sắp xếp kết quả:} Theo giá tăng dần/giảm dần, diện tích tăng dần/giảm dần, mới nhất.
    \end{itemize}

    \item \textbf{Xem chi tiết bất động sản:}
    \begin{itemize}
        \item Trang chi tiết hiển thị đầy đủ thông tin của một bất động sản: tiêu đề, mô tả chi tiết, giá (dạng số + text), đơn giá theo m\textsuperscript{2}, diện tích (dạng số + text), địa chỉ đầy đủ (phường/xã, quận/huyện, tỉnh/thành phố), tọa độ địa lý (latitude, longitude).
        \item Thông số kỹ thuật: số phòng ngủ, số phòng tắm, số tầng, hướng nhà, hướng ban công, mặt tiền, đường vào.
        \item Thông tin pháp lý và nội thất: tình trạng pháp lý (sổ đỏ, sổ hồng,\dots), tình trạng nội thất (đầy đủ, cơ bản, bàn giao thô).
        \item Thông tin liên hệ: tên và số điện thoại người đăng.
        \item \textbf{Bất động sản tương tự:} Hệ thống tự động gợi ý các bất động sản có đặc điểm tương đồng (cùng khu vực, cùng phân khúc giá, cùng loại hình).
    \end{itemize}

    \item \textbf{Thống kê và phân tích thị trường:}
    \begin{itemize}
        \item \textbf{Tổng quan thị trường:} Hiển thị các chỉ số chính: tổng số tin đăng đang hoạt động, giá trung bình, diện tích trung bình, số lượng tin bán và cho thuê, số thành phố và quận/huyện có dữ liệu.
        \item \textbf{Top khu vực:} Bảng xếp hạng các thành phố/quận có nhiều tin đăng nhất, có thể lọc theo loại tin.
        \item \textbf{Thống kê giá theo quận:} Với mỗi quận trong một thành phố, hiển thị số lượng tin, giá trung bình, giá thấp nhất, giá cao nhất và đơn giá trung bình/m\textsuperscript{2}.
        \item \textbf{Phân bố loại hình:} Thống kê số lượng tin theo từng loại hình bất động sản (căn hộ, nhà riêng, đất nền,\dots).
    \end{itemize}

    \item \textbf{Tương tác với chatbot tư vấn:}
    \begin{itemize}
        \item \textbf{Chat widget nổi:} Luôn hiển thị ở góc phải dưới màn hình, có thể mở rộng/thu gọn. Người dùng có thể đặt câu hỏi bất kỳ lúc nào.
        \item \textbf{Hội thoại đa lượt:} Hệ thống lưu trữ toàn bộ lịch sử hội thoại theo phiên (session). Người dùng có thể xem lại các phiên chat trước đó.
        \item \textbf{Phản hồi có trích dẫn nguồn:} Mỗi câu trả lời của chatbot đều kèm danh sách nguồn tham khảo (tên bất động sản, đường dẫn, điểm liên quan), cho phép người dùng kiểm chứng thông tin.
        \item \textbf{Gợi ý câu hỏi tiếp theo:} Sau mỗi phản hồi, chatbot đề xuất 2--3 câu hỏi liên quan để người dùng tiếp tục khai thác.
        \item \textbf{Đánh giá chất lượng:} Người dùng có thể đánh giá phản hồi (thích/không thích) và để lại nhận xét, giúp cải thiện chất lượng hệ thống.
        \item \textbf{Cá nhân hóa:} Hệ thống ghi nhớ sở thích của người dùng (khu vực ưa thích, ngân sách, loại hình) và sử dụng chúng để cá nhân hóa phản hồi trong các phiên sau.
    \end{itemize}
\end{enumerate}

\subsubsection{Nhóm chức năng chatbot đa tác tử}

Đây là nhóm chức năng cốt lõi tạo nên sự khác biệt của hệ thống so với các nền tảng bất động sản truyền thống:

\begin{enumerate}
    \item \textbf{Phân loại ý định người dùng (Intent Routing):}
    \begin{itemize}
        \item Hệ thống tự động phân tích câu hỏi tiếng Việt để xác định người dùng đang cần: tìm kiếm bất động sản, phân tích thị trường, tư vấn pháp lý, hay tư vấn đầu tư.
        \item \textbf{Hai cơ chế routing:} (1) Keyword-based -- sử dụng regex và từ khóa tiếng Việt không dấu để nhận diện ý định, hoạt động không cần API key; (2) Gemini-based -- sử dụng Google Gemini 2.0 Flash để phân loại với độ chính xác cao hơn, tự động kích hoạt khi có GEMINI\_API\_KEY.
        \item Router có thể chọn nhiều agent cùng lúc nếu câu hỏi liên quan đến nhiều lĩnh vực (ví dụ: ``căn hộ Quận 7 dưới 3 tỷ, thủ tục pháp lý thế nào'' sẽ kích hoạt cả Property Search và Legal Advisor).
    \end{itemize}

    \item \textbf{Tìm kiếm bất động sản thông minh (Property Search):}
    \begin{itemize}
        \item Tự động trích xuất bộ lọc có cấu trúc từ câu hỏi tiếng Việt: loại giao dịch (mua/thuê), loại hình (căn hộ, nhà,\dots), thành phố, quận, khoảng giá, diện tích. Ví dụ: ``tìm căn hộ Quận 7 dưới 5 tỷ, 2 phòng ngủ'' được parse thành \texttt{\{listing\_type: sale, property\_type: apartment, district: Quận 7, max\_price: 5, bedrooms: 2\}}.
        \item Thực hiện hybrid search kết hợp SQL filtering và pgvector semantic search. Kết quả được xếp hạng lại qua Cohere Rerank (nếu có API key) và trả về kèm giải thích lý do từng bất động sản phù hợp.
    \end{itemize}

    \item \textbf{Phân tích thị trường (Market Analysis):}
    \begin{itemize}
        \item Cung cấp số liệu thống kê thị trường theo thời gian thực: giá trung bình, giá/m\textsuperscript{2}, số lượng tin đăng, xu hướng biến động theo khu vực và loại hình.
        \item So sánh giữa các khu vực: giá trung bình tại Quận 7 so với Quận 2, giá căn hộ so với nhà riêng trong cùng khu vực.
    \end{itemize}

    \item \textbf{Tư vấn pháp lý (Legal Advisor):}
    \begin{itemize}
        \item Cung cấp thông tin về các vấn đề pháp lý thường gặp trong giao dịch bất động sản: sổ đỏ/sổ hồng, thủ tục sang tên, công chứng hợp đồng, các loại thuế và phí, quy hoạch, điều kiện vay ngân hàng.
        \item Thông tin được truy xuất từ Legal Knowledge Base -- kho tài liệu pháp lý được xử lý và lập chỉ mục định kỳ hàng tháng.
    \end{itemize}

    \item \textbf{Tư vấn đầu tư (Investment Advisor):}
    \begin{itemize}
        \item Phân tích tiềm năng đầu tư dựa trên các yếu tố: vị trí, giá, xu hướng thị trường, tiện ích xung quanh, quy hoạch hạ tầng.
        \item Đánh giá sơ bộ về thanh khoản, khả năng cho thuê lại, tỷ suất sinh lời kỳ vọng và biên an toàn tài chính.
    \end{itemize}

    \item \textbf{Tổng hợp và phản hồi (Synthesis):}
    \begin{itemize}
        \item Khi nhiều agent cùng được kích hoạt, orchestrator gọi chúng song song và tổng hợp kết quả thành một phản hồi duy nhất, mạch lạc, tránh trùng lặp.
        \item Mỗi phản hồi đều được kiểm tra an toàn (safety validation) trước khi gửi đến người dùng.
    \end{itemize}

    \item \textbf{Ghi nhớ và cá nhân hóa (Memory System):}
    \begin{itemize}
        \item Chatbot tự động trích xuất sở thích người dùng từ hội thoại (ví dụ: ``Tôi thích căn hộ ở Quận 7, ngân sách khoảng 3 tỷ'') và lưu dưới dạng MemoryProposal.
        \item Các sở thích có độ tin cậy cao ($\ge 80\%$) được tự động áp dụng. Các sở thích khác cần người dùng xác nhận.
        \item Sở thích đã xác nhận được sử dụng để cá nhân hóa phản hồi trong các phiên chat sau.
    \end{itemize}
\end{enumerate}

\subsubsection{Nhóm chức năng quản trị và tự động hóa}

\begin{enumerate}
    \item \textbf{Thu thập dữ liệu tự động:}
    \begin{itemize}
        \item \textbf{Crawl tin đăng hàng ngày:} Tự động thu thập tin bán và cho thuê mới từ batdongsan.com.vn vào 02:00 giờ Hồ Chí Minh. Cơ chế watermark đảm bảo chỉ crawl dữ liệu mới, tránh trùng lặp.
        \item \textbf{Crawl dự án hàng tuần:} Thu thập thông tin dự án bất động sản (chủ đầu tư, quy mô, vị trí, tiện ích) vào Chủ nhật hàng tuần.
        \item \textbf{Crawl tin tức hàng tuần:} Thu thập bài viết phân tích thị trường, tin tức bất động sản.
        \item \textbf{Cập nhật kiến thức pháp lý hàng tháng:} Xử lý lại toàn bộ tài liệu pháp lý, bao gồm cả HTML và PDF.
    \end{itemize}

    \item \textbf{Xử lý dữ liệu (ETL Pipeline):}
    \begin{itemize}
        \item \textbf{Làm sạch:} Chuẩn hóa giá (tỷ, triệu, triệu/tháng), diện tích (m\textsuperscript{2}), phân loại loại hình bất động sản, chuẩn hóa địa chỉ.
        \item \textbf{Làm giàu:} Geocoding -- chuyển địa chỉ thành tọa độ (latitude, longitude) qua Nominatim hoặc Goong API. Intent tagging -- tự động gán nhãn nhu cầu (``gần trường học'', ``view đẹp'', ``an ninh'',\dots) dựa trên nội dung mô tả.
        \item \textbf{Chunking:} Chia văn bản thành các đoạn ngữ nghĩa (chunks) với 4 loại: overview (tổng quan), description (mô tả chi tiết), location (vị trí), intent\_tags (nhu cầu). Mỗi chunk được thiết kế để có thể đứng độc lập về mặt ngữ nghĩa.
        \item \textbf{Embedding:} Tạo vector embedding 1024 chiều cho từng chunk sử dụng mô hình BGE-M3 (BAAI/bge-m3) với batch size 16 và retry policy 3 lần.
        \item \textbf{Nạp dữ liệu:} Upsert vào PostgreSQL: bảng listings cho dữ liệu quan hệ, bảng chunks cho dữ liệu vector. Sử dụng cơ chế batch processing để tối ưu hiệu năng.
    \end{itemize}

    \item \textbf{Quản trị và giám sát hệ thống:}
    \begin{itemize}
        \item \textbf{Admin Dashboard:} Giao diện quản trị với các tab: Pipeline Readiness (trạng thái các nguồn dữ liệu), Agent Traces (lịch sử xử lý của agent với latency và status), Chat Feedback (phản hồi người dùng), Memory Proposals (các đề xuất đang chờ xác nhận).
        \item \textbf{Observability:} Hệ thống tracing toàn diện cho phép theo dõi toàn bộ luồng xử lý của từng request chatbot: intent, agents được chọn, thời gian xử lý từng bước, kết quả retrieval, LLM calls, errors.
        \item \textbf{Monitoring:} Prometheus metrics endpoint cung cấp các chỉ số: số lượng request chat, thời gian phản hồi, số lượng pipeline runs, tỷ lệ lỗi.
        \item \textbf{Alerting:} Cơ chế cảnh báo qua Slack và email khi pipeline thất bại hoặc hiệu năng bất thường.
    \end{itemize}
\end{enumerate}

        \item Database connection pool: 20 connections + 10 overflow

    Bảo mật &
    \begin{itemize}
        \item Xác thực JWT với HS256, token hết hạn sau 24h

        \item Chống SQL injection qua SQLAlchemy ORM (parameterized queries)
        \item Chống XSS qua Next.js auto-escaping
        \item CORS whitelist qua biến môi trường CORS\_ORIGINS
        \item Internal API giữa backend và agent service bảo vệ bằng internal key
    \end{itemize} \\
    \hline
    Độ tin cậy &
    \begin{itemize}

        \item Retry policy cho Airflow tasks: 3 lần, 5--30 phút
        \item Health check endpoints cho tất cả services
        \item Database health check trước khi khởi động service
    \end{itemize} \\
    \hline
    Khả năng bảo trì &
    \begin{itemize}
        \item Mã nguồn phân chia module rõ ràng theo domain
        \item Docstring tiếng Anh cho tất cả hàm/class
        \item Type hints đầy đủ (Python) và TypeScript (frontend)
        \item Hơn 55 file test bao phủ toàn bộ module
        \item ESLint cho frontend, compileall cho Python
    \end{itemize} \\

    Tương thích ngôn ngữ &
    \begin{itemize}
        \item Giao diện người dùng: 100\% tiếng Việt
        \item Chatbot: hiểu và phản hồi bằng tiếng Việt
        \item Embedding: BGE-M3 tối ưu cho hơn 100 ngôn ngữ bao gồm tiếng Việt
        \item URL slugs: tiếng Việt không dấu
    \end{itemize} \\

\subsubsection{Tầng Người dùng (User Layer)}

Người dùng tương tác với hệ thống qua trình duyệt web (desktop và mobile). Có hai nhóm người dùng chính:
\begin{itemize}
    \item \textbf{Người dùng cuối:} Truy cập các trang duyệt bất động sản, xem thống kê thị trường và tương tác với chatbot tư vấn.
    \item \textbf{Quản trị viên:} Truy cập admin dashboard để giám sát pipeline, theo dõi agent traces và quản lý phản hồi người dùng.
\end{itemize}

\subsubsection{Tầng Frontend -- Next.js (Port 3000)}

Frontend được xây dựng với Next.js 16 và React 19, sử dụng App Router để tổ chức các trang. Các đặc điểm thiết kế chính:
\begin{itemize}
    \item \textbf{Server-Side Rendering (SSR):} Các trang danh sách và chi tiết bất động sản được render phía server, đảm bảo SEO và thời gian tải nhanh.
    \item \textbf{React Server Components (RSC):} Mặc định sử dụng Server Components để giảm JavaScript gửi về client.
    \item \textbf{Tailwind CSS v4:} Utility-first CSS framework đảm bảo giao diện nhất quán, responsive trên mọi thiết bị.
    \item \textbf{Typed API Client:} File \texttt{lib/api.ts} cung cấp các hàm gọi backend với TypeScript type safety, tự động gắn JWT token vào header.
    \item \textbf{Chat Widget:} Thành phần nổi (floating) luôn hiển thị, cho phép người dùng tương tác với chatbot ở bất kỳ trang nào.
\end{itemize}

Frontend chỉ giao tiếp trực tiếp với Backend API qua REST. Frontend không được phép gọi trực tiếp Agent Service -- đây là nguyên tắc thiết kế quan trọng để đảm bảo bảo mật và khả năng bảo trì.

\subsubsection{Tầng Backend API -- FastAPI (Port 8000)}

Backend API là cổng giao tiếp chính của toàn bộ hệ thống, đảm nhận mọi request từ frontend. Backend được tổ chức thành 4 lớp:

\begin{enumerate}
    \item \textbf{Lớp Models (ORM):} Định nghĩa 14+ bảng dữ liệu với SQLAlchemy 2.0 async. Mỗi model ánh xạ trực tiếp đến một bảng trong PostgreSQL. Các mối quan hệ (relationships) được định nghĩa rõ ràng: Listing $\rightarrow$ Chunk (polymorphic), User $\rightarrow$ ChatSession $\rightarrow$ ChatMessage, User $\rightarrow$ UserPreference.

    \item \textbf{Lớp Schemas (Pydantic):} Định nghĩa contract cho request/response. Pydantic v2 tự động validate dữ liệu đầu vào, sinh JSON Schema và tài liệu OpenAPI. Các schema được thiết kế tách biệt cho từng nghiệp vụ: listing (ListingCard, ListingDetail, PaginatedResponse), chat (ChatMessageRequest, ChatMessageResponse), auth (UserRegisterRequest, TokenResponse), admin (TraceSummary, PipelineReadinessItem).

    \item \textbf{Lớp Bộ định tuyến (API Endpoints):} Xử lý yêu cầu HTTP, gọi dịch vụ tương ứng và trả về phản hồi. Máy chủ có 8 mô-đun định tuyến, mỗi mô-đun phụ trách một nhóm chức năng: bất động sản, thị trường, trò chuyện, xác thực, sở thích, ghi nhớ, quản trị và chỉ số. Các bộ định tuyến sử dụng tiêm phụ thuộc để nhận phiên cơ sở dữ liệu và xác thực người dùng.

    \item \textbf{Lớp Services (Business Logic):} Chứa toàn bộ logic nghiệp vụ, được chia thành 3 nhóm chính:
    \begin{itemize}
        \item \textbf{Chatbot Pipeline (\texttt{services/chatbot/}):} Pipeline multi-agent nội bộ với router (phân loại ý định), orchestrator (điều phối và tổng hợp), 4 agent chuyên biệt, context builder (xây dựng ngữ cảnh hội thoại), memory manager (quản lý ghi nhớ người dùng).
        \item \textbf{RAG Services (\texttt{services/rag/}):} Các dịch vụ truy xuất thông tin: hybrid search (kết hợp SQL + vector + rerank + cache), embeddings (BGE-M3 wrapper với batch processing và retry), cache (Redis caching cho embedding và kết quả tìm kiếm).
        \item \textbf{Agent Service Client (\texttt{services/agent\_service/}):} HTTP client giao tiếp với Agent Service qua internal API, sử dụng internal key authentication.
    \end{itemize}
\end{enumerate}

\subsubsection{Tầng Agent Service -- FastAPI + LangGraph (Port 8100)}

Agent Service là dịch vụ vi mô riêng biệt, được thiết kế để xử lý các tác vụ trí tuệ nhân tạo phức tạp mà không ảnh hưởng đến hiệu năng của Máy chủ Dịch vụ chính. Dịch vụ này sử dụng LangGraph để điều phối quy trình đa tác tử qua đồ thị trạng thái 14 nút, tổ chức thành bốn nhóm chức năng.

Nhóm chuẩn bị và định tuyến gồm năm nút. Nút xây dựng ngữ cảnh tổng hợp lịch sử trò chuyện và sở thích người dùng để đưa vào lời nhắc cho tất cả các bước sau. Nút kiểm tra trạng thái sẵn sàng đánh giá tính khả dụng của từng nguồn dữ liệu (bất động sản, dự án, bài viết, kho kiến thức pháp lý) và điều chỉnh chiến lược truy xuất khi có nguồn chưa sẵn sàng. Nút định tuyến phân loại ý định và chọn danh sách tác tử cần kích hoạt, sử dụng Gemini 2.0 Flash với lời nhắc thiết kế riêng cho lĩnh vực bất động sản tiếng Việt. Nút làm rõ phát hiện các truy vấn thiếu thông tin và tạo câu hỏi làm rõ trước khi tiếp tục xử lý. Nút hiểu truy vấn phân tích sâu ngữ nghĩa để trích xuất thực thể, ràng buộc và ngữ cảnh ẩn.

Nhóm truy xuất và xử lý chuyên biệt gồm ba nút. Nút lập kế hoạch truy xuất xác định nguồn dữ liệu cần truy vấn, bộ lọc áp dụng và độ ưu tiên, kết quả được cấu trúc thành danh sách nhiệm vụ truy xuất. Nút tác tử chuyên biệt thực thi đồng thời các tác tử được chọn: tìm kiếm bất động sản (thực hiện tìm kiếm lai và trích xuất bộ lọc), phân tích thị trường (truy vấn thống kê và xu hướng), tư vấn pháp lý (tìm trong kho kiến thức pháp lý) và tư vấn đầu tư (phân tích tiềm năng sinh lời). Nút mô hình đầu tư tính toán các chỉ số tài chính cho bất động sản được tư vấn đầu tư.

Nhóm tổng hợp và kiểm tra gồm hai nút. Nút tổng hợp kết hợp kết quả từ các tác tử thành phản hồi mạch lạc, ưu tiên thông tin có độ tin cậy cao và loại bỏ trùng lặp. Nút kiểm tra an toàn đánh giá phản hồi theo bốn tiêu chí: tính có căn cứ (dựa trên nguồn dữ liệu thực), tính hữu ích, an toàn (không đưa ra khẳng định pháp lý hay tài chính tuyệt đối) và chất lượng trích dẫn.

Nhóm phản ứng và cải thiện gồm ba nút. Nút điều khiển phản ứng quyết định có cần truy xuất thêm hoặc chạy lại tác tử khi phát hiện phản hồi thiếu bằng chứng. Nút truy xuất phản ứng thực hiện truy vấn bổ sung theo yêu cầu. Nút đánh giá hội đồng cho phép các tác tử xem xét chéo kết quả của nhau. Cuối cùng, nút đề xuất ghi nhớ trích xuất sở thích tiềm năng của người dùng từ nội dung hội thoại và tạo đề xuất để lưu vào hệ thống.

\textbf{Cơ chế đánh giá chất lượng:}

Sau mỗi phản hồi, hệ thống có thể kích hoạt bộ đánh giá tự động sử dụng Gemini để chấm điểm theo năm tiêu chí (thang 0,0--1,0): tính có căn cứ, tính hữu ích, chất lượng trích dẫn, an toàn và tính đầy đủ của vết xử lý. Kết quả đánh giá được lưu vào bảng vết tác tử để phân tích và cải thiện hệ thống.

\textbf{Tương tác giữa Máy chủ Dịch vụ và Dịch vụ Tác tử:}

Máy chủ Dịch vụ gọi Dịch vụ Tác tử qua trình khách HTTP với xác thực bằng khóa nội bộ. Dịch vụ Tác tử nhận yêu cầu chứa câu hỏi, mã phiên, mã người dùng và lịch sử hội thoại, trả về phản hồi cuối cùng, danh sách tác tử đã sử dụng, nguồn tham khảo, hành động gợi ý và vết xử lý đầy đủ. Máy chủ Dịch vụ lưu kết quả vào các bảng tin nhắn, vết tác tử và bảng liên quan trước khi trả về giao diện người dùng. Khi Dịch vụ Tác tử không được kích hoạt hoặc không khả dụng, Máy chủ Dịch vụ tự động chuyển sang đường ống nội bộ.

\subsubsection{Tầng Dữ liệu}

\textbf{PostgreSQL 16 với tìm kiếm véc-tơ (cổng 5432):}

PostgreSQL được chọn làm cơ sở dữ liệu chính nhờ khả năng lưu trữ đồng thời dữ liệu quan hệ và dữ liệu véc-tơ thông qua phần mở rộng pgvector. Điều này giúp giảm độ phức tạp vận hành so với việc sử dụng hai hệ quản trị riêng biệt.

Dữ liệu quan hệ gồm 14 bảng: bất động sản (30 trường), người dùng, phiên và tin nhắn trò chuyện, dự án, bài viết, cùng các bảng giám sát (vết tác tử, các bước vết, lời gọi mô hình ngôn ngữ, sự kiện truy xuất, lần chạy đường ống, trạng thái nguồn, sở thích người dùng, đề xuất ghi nhớ, phản hồi đánh giá).

Dữ liệu véc-tơ được lưu trong bảng đoạn văn bản với cột véc-tơ 1024 chiều, phục vụ tìm kiếm ngữ nghĩa. Chỉ mục HNSW với tham số $m=16$ và $ef\_construction=64$ đảm bảo tốc độ tìm kiếm k lân cận gần nhất nhanh trên tập dữ liệu lớn. Thiết kế đa hình cho phép một bảng đoạn văn bản duy nhất phục vụ nhiều loại dữ liệu nguồn (bất động sản, dự án, bài viết) thông qua cặp cột loại nguồn và mã nguồn.

\textbf{Redis 7 (cổng 6379):}

Redis được sử dụng làm bộ nhớ đệm ở hai cấp độ: bộ nhớ đệm véc-tơ đặc trưng (lưu véc-tơ của câu truy vấn, tránh mã hóa lại) và bộ nhớ đệm kết quả tìm kiếm (lưu kết quả tìm kiếm lai cho truy vấn lặp lại).
\begin{itemize}
    \item \textbf{Embedding Cache:} Lưu vector embedding của câu truy vấn, tránh phải encode lại. Key có dạng \texttt{embedding:\{provider\}:\{model\}:\{dim\}:\{hash\}}, cho phép cache hoạt động chính xác ngay cả khi thay đổi model.
    \item \textbf{Search Result Cache:} Lưu kết quả hybrid search cho các truy vấn lặp lại, giảm tải đáng kể cho PostgreSQL.
\end{itemize}

\subsubsection{Tầng Data Pipeline \& Tự động hóa}

Đây là tầng vận hành ngầm, đảm bảo dữ liệu luôn được cập nhật và sẵn sàng cho các tầng phía trên.

\textbf{Crawlers (Playwright + Stealth):}

Hệ thống crawler được tổ chức thành 5 module, mỗi module phụ trách một loại nội dung:
\begin{itemize}
    \item \texttt{crawler/core/}: Các thành phần dùng chung -- CSV writer (với dedup), base parser, listing detail parser.
    \item \texttt{crawler/sale/}: Crawl tin bán (2 bước: crawl URLs $\rightarrow$ crawl details).
    \item \texttt{crawler/rent/}: Crawl tin cho thuê (tương tự).
    \item \texttt{crawler/projects/}: Crawl thông tin dự án bất động sản.
    \item \texttt{crawler/news/}: Crawl tin tức và bài viết phân tích.
\end{itemize}

Crawler sử dụng Playwright với chế độ stealth để tránh bị phát hiện, hỗ trợ 4 workers chạy song song để tăng tốc độ. Dữ liệu được ghi ra file CSV UTF-8 BOM và được deduplicate dựa trên product\_id.

\textbf{Data Pipeline (ETL + Embedding):}

Pipeline xử lý dữ liệu gồm 5 bước chính, được thiết kế để chạy độc lập hoặc theo chuỗi:

\begin{enumerate}
    \item \textbf{Clean (\texttt{clean.py}):} Parse và chuẩn hóa dữ liệu thô từ CSV. Các công việc chính: tách giá trị số từ text giá (``4,68 tỷ'' $\rightarrow$ 4.68, unit: billion), tách diện tích (``75 m\textsuperscript{2}'' $\rightarrow$ 75), phân loại loại hình bất động sản, chuẩn hóa địa chỉ.
    
    \item \textbf{Enrich (\texttt{enrich.py}):} Làm giàu dữ liệu với hai tác vụ: (1) Geocoding -- gọi Nominatim/Goong API để chuyển địa chỉ text thành tọa độ (latitude, longitude); (2) Intent Tagging -- phân tích nội dung mô tả để gán nhãn nhu cầu (``gần trường học'', ``view đẹp'', ``an ninh'', ``pháp lý rõ'',\dots).
    
    \item \textbf{Chunk (\texttt{chunk.py}):} Chia văn bản thành các đoạn ngữ nghĩa. Mỗi listing được chunk thành 4 loại: overview (tổng quan -- tiêu đề, giá, diện tích, khu vực, pháp lý, nội thất), description (mô tả chi tiết), location (địa chỉ, quận, thành phố), intent\_tags (nhu cầu phù hợp). Thiết kế này đảm bảo mỗi chunk có thể đứng độc lập và được truy xuất chính xác theo ngữ cảnh.
    
    \item \textbf{Embed (\texttt{embed.py}):} Tạo vector embedding 1024 chiều cho từng chunk bằng mô hình BGE-M3. Sử dụng BGEEmbedder class với batch size 16, retry 3 lần với exponential backoff, chạy async qua \texttt{asyncio.to\_thread()} để không chặn event loop.
    
    \item \textbf{Load (\texttt{load\_db.py} + ingestors/):} Upsert dữ liệu đã xử lý vào PostgreSQL. Các ingestor module chuyên biệt cho từng loại dữ liệu (listings, projects, news, legal\_kb) đảm bảo logic nạp phù hợp với schema của từng bảng.
\end{enumerate}

\textbf{Apache Airflow Scheduler:}

Toàn bộ pipeline được lập lịch và giám sát bởi Apache Airflow với 4 DAGs:

\begin{itemize}
    \item \textbf{daily\_listings\_dag (02:00 HCM):} Quy trình đầy đủ cho tin đăng: crawl\_sale\_urls $\rightarrow$ crawl\_sale\_details $\rightarrow$ ingest\_sale (chạy song song với crawl\_rent\_urls $\rightarrow$ crawl\_rent\_details $\rightarrow$ ingest\_rent) $\rightarrow$ mark\_active $\rightarrow$ chunk\_and\_embed. Sử dụng watermark để chỉ xử lý dữ liệu mới.
    \item \textbf{weekly\_projects\_dag (Chủ nhật 03:00):} Crawl và nạp dự án bất động sản.
    \item \textbf{weekly\_news\_dag (Chủ nhật 04:00):} Crawl và nạp tin tức thị trường.
    \item \textbf{monthly\_legal\_kb\_dag (Ngày 1, 05:00):} Xử lý lại toàn bộ tài liệu pháp lý từ \texttt{data/knowledge/}.
\end{itemize}

Mỗi DAG được cấu hình retry 3 lần với exponential backoff (5--30 phút), cảnh báo qua Slack và email khi thất bại, và ghi metrics vào bảng pipeline\_runs.

\section{Thiết kế cơ sở dữ liệu}

\subsection{Mô hình quan hệ}

Hệ thống sử dụng PostgreSQL 16 với extension pgvector. Mô hình dữ liệu được thiết kế xoay quanh 3 nhóm bảng chính:

\subsubsection{Nhóm bảng nghiệp vụ (Business Tables)}

Đây là nhóm bảng lưu trữ dữ liệu cốt lõi của hệ thống:

\begin{itemize}
    \item \textbf{listings -- Bảng bất động sản:} Bảng quan trọng nhất với hơn 30 trường dữ liệu, lưu toàn bộ thông tin của một tin đăng. Khóa chính là id (INTEGER, auto-increment). Mỗi listing có product\_id duy nhất từ nguồn crawl để deduplicate. Các chỉ mục (indexes) được tạo trên các trường thường xuyên dùng trong bộ lọc: listing\_type, city, district, property\_type, price, area, bedrooms, post\_date, is\_active.

    \item \textbf{projects -- Bảng dự án:} Lưu thông tin dự án bất động sản: tên, chủ đầu tư, vị trí, quy mô (tổng số căn), khoảng giá, khoảng diện tích, trạng thái (sắp mở bán, đang bán, đã bàn giao), loại hình, tiện ích (mảng ARRAY), mô tả.

    \item \textbf{articles -- Bảng bài viết:} Lưu tin tức và kiến thức pháp lý: tiêu đề, nội dung, danh mục, nguồn, ngày đăng, URL gốc, metadata.

    \item \textbf{users -- Bảng người dùng:} Lưu thông tin tài khoản: email (unique), mật khẩu đã hash, họ tên, số điện thoại, avatar, trạng thái hoạt động, quyền (admin/superuser).
\end{itemize}

\subsubsection{Nhóm bảng hội thoại (Chat Tables)}

\begin{itemize}
    \item \textbf{chat\_sessions -- Phiên hội thoại:} Mỗi phiên có id UUID, liên kết đến user (có thể NULL cho anonymous), tiêu đề tự động sinh từ tin nhắn đầu tiên.
    \item \textbf{chat\_messages -- Tin nhắn:} Mỗi tin nhắn thuộc về một session, có role (user/assistant/system), nội dung, agent đã xử lý, và metadata JSONB (chứa sources, citations, search context).
\end{itemize}

\subsubsection{Nhóm bảng vector và tìm kiếm (Vector \& Retrieval Tables)}

\begin{itemize}
    \item \textbf{chunks -- Đoạn văn bản đã embedding:} Thiết kế polymorphic với parent\_type (listing/project/article) và parent\_id. Mỗi chunk có chunk\_type xác định loại nội dung (overview/description/location/intent\_tags/legal), text là nội dung văn bản, embedding là VECTOR(1024). HNSW index trên cột embedding với cosine distance operator đảm bảo tìm kiếm k-NN nhanh.
\end{itemize}

\subsubsection{Nhóm bảng observability \& vận hành (Observability Tables)}

\begin{itemize}
    \item \textbf{agent\_traces:} Lưu vết xử lý của mỗi request chatbot: intent, danh sách agents đã dùng, trace\_summary, full\_trace (JSONB), latency, status, graph\_version, prompt\_version, model\_name.
    \item \textbf{agent\_trace\_steps:} Chi tiết từng bước trong trace: step\_name, status, latency, input/output JSON.
    \item \textbf{agent\_llm\_calls:} Log mỗi lần gọi LLM: node\_name, model\_name, latency, token estimates.
    \item \textbf{agent\_retrieval\_events:} Log mỗi sự kiện truy xuất.
    \item \textbf{pipeline\_runs:} Log mỗi lần chạy Airflow DAG: dag\_id, run\_id, status, thời gian, metrics.
    \item \textbf{source\_readiness:} Trạng thái sẵn sàng của từng nguồn dữ liệu.
    \item \textbf{user\_preferences:} Sở thích người dùng dạng key-value JSONB.
    \item \textbf{memory\_proposals:} Đề xuất sở thích từ agent.
    \item \textbf{chat\_feedback:} Phản hồi của người dùng về chất lượng chat.
\end{itemize}

\subsection{Chiến lược indexing}

\begin{itemize}
    \item \textbf{Index cho listings:} Các B-tree index trên listing\_type, city, district, property\_type, price, area, bedrooms, post\_date, is\_active giúp tối ưu các truy vấn lọc và sắp xếp phổ biến. Composite index ngầm được PostgreSQL sử dụng khi nhiều điều kiện AND.
    \item \textbf{Index cho chunks:} HNSW index trên cột embedding với vector\_cosine\_ops. Đây là loại index xấp xỉ (approximate nearest neighbor) cho phép tìm kiếm k-NN trong không gian 1024 chiều với độ phức tạp $O(\log N)$ thay vì $O(N)$ của full scan. Tham số $m=16$ (số kết nối mỗi node trong đồ thị HNSW) và $ef\_construction=64$ (kích thước dynamic candidate list) được chọn sau thử nghiệm để cân bằng giữa tốc độ build index và chất lượng tìm kiếm.
    \item \textbf{Index cho polymorphic queries:} Composite index trên (parent\_type, parent\_id) của bảng chunks giúp truy vấn nhanh tất cả chunks thuộc về một listing/project/article.
\end{itemize}
